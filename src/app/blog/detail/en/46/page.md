## 高阶组件

```javascript
/**
 * 高阶组件必须要是一个纯函数
 * 高阶组件会把传给自己的props给构建时的参数组件
 * 高阶组件会返回一个新的组件，其包含参数组件
 */

// 高阶组件一般使用with前缀命名
const withDoNothing = (component) => {
  const newComponent = (props) => {
    return <Component {...props} />
  };
  return newComponent;
}

/// compose 实现
function compose(...funcs) {
  if(funcs,length ===0 ){
    return arg => arg;
  }
  if(funcs.length === 1) {
    return funcs[0];
  }

  return funcs.reduce((a,b) => (...args) => a(b(...args)));
}
```
